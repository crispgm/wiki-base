Change HDD

# mount && swapon
mount /dev/sda3 /mnt # replace with nvmen1p3 and etc.
swapon /dev/sda2

# delete old fstab files
echo "" > /mnt/etc/fstab

# genfstab again
genfstab -U /mnt >> /mnt/etc/fstab
